    <script>

        // Gebruik unieke keys om conflicten te voorkomen

        const STORAGE_KEYS = {

            DISHES: 'afiyet_dishes_v10',

            PLANS: 'afiyet_plans_v10',

            EXTRAS: 'afiyet_extras_v10'

        };


        let currentWeekOffset = 0;

        let dishes = JSON.parse(localStorage.getItem(STORAGE_KEYS.DISHES)) || [

            { id: 1, name: 'Menemen', category: 'Ontbijt' },

            { id: 2, name: 'Linzensoep', category: 'Lunch' },

            { id: 3, name: 'Pasta', category: 'Diner' }

        ];

        let allWeekPlans = JSON.parse(localStorage.getItem(STORAGE_KEYS.PLANS)) || { "-1": {}, "0": {}, "1": {} };

        let extras = JSON.parse(localStorage.getItem(STORAGE_KEYS.EXTRAS)) || [];


        function init() {

            renderFullDishGrid();

            updateWeekUI();

            renderShoppingList();

        }


        // --- CORE FUNCTIES ---

        function saveToStorage() {

            localStorage.setItem(STORAGE_KEYS.DISHES, JSON.stringify(dishes));

            localStorage.setItem(STORAGE_KEYS.PLANS, JSON.stringify(allWeekPlans));

            localStorage.setItem(STORAGE_KEYS.EXTRAS, JSON.stringify(extras));

        }


        function switchTab(tab) {

            ['planner', 'recipes', 'shopping'].forEach(t => {

                document.getElementById(`section-${t}`).classList.toggle('hidden', t !== tab);

                document.getElementById(`tab-${t}`).classList.toggle('active', t === tab);

            });

            if (tab === 'shopping') renderShoppingList();

        }


        function changeWeek(offset) {

            currentWeekOffset = offset;

            document.querySelectorAll('.week-btn').forEach(btn => btn.classList.remove('active'));

            if(offset === -1) document.getElementById('btn-prev').classList.add('active');

            if(offset === 0) document.getElementById('btn-current').classList.add('active');

            if(offset === 1) document.getElementById('btn-next').classList.add('active');

            updateWeekUI();

        }


        function handleExport() {

            // Forceer een korte pauze voor de browser om UI te updaten voor het printvenster

            setTimeout(() => {

                window.print();

            }, 250);

        }


        // --- UI RENDERING ---

        function updateWeekUI() {

            const grid = document.getElementById('weekGrid');

            grid.innerHTML = '';

            const plan = allWeekPlans[currentWeekOffset] || {};

            const days = ['Maandag', 'Dinsdag', 'Woensdag', 'Donderdag', 'Vrijdag', 'Zaterdag', 'Zondag'];

            const cats = ['Ontbijt', 'Lunch', 'Diner'];


            days.forEach(day => {

                let catHtml = '';

                cats.forEach(cat => {

                    const saved = plan[`${day}-${cat}`] || '';

                    const options = dishes.filter(d => d.category === cat).map(d => 

                        `<option value="${d.name}" ${saved === d.name ? 'selected' : ''}>${d.name}</option>`

                    ).join('');


                    catHtml += `

                        <div class="p-3 border-b border-orange-100 last:border-0 hover:bg-orange-50 transition">

                            <span class="text-[10px] uppercase font-black text-orange-300 block mb-1">${cat}</span>

                            <select onchange="savePlan(this.value, '${day}', '${cat}')" class="w-full bg-transparent text-[13px] font-bold text-gray-700 outline-none">

                                <option value="">-- Kies --</option>

                                ${options}

                                <option value="">(Leeg)</option>

                            </select>

                            <div class="mt-1 text-[11px] font-bold text-primary truncate h-4">${saved}</div>

                        </div>`;

                });


                grid.innerHTML += `

                    <div class="day-card flex flex-col bg-warm-cream border border-orange-100 rounded-2xl shadow-sm overflow-hidden min-h-[250px]">

                        <div class="bg-primary text-white text-center py-2 text-sm font-bold uppercase">${day}</div>

                        ${catHtml}

                    </div>`;

            });


            const labels = { "-1": "Vorige week", "0": "Deze week", "1": "Volgende week" };

            document.getElementById('currentWeekTitle').innerText = labels[currentWeekOffset];

        }


        window.savePlan = (val, day, cat) => {

            if(!allWeekPlans[currentWeekOffset]) allWeekPlans[currentWeekOffset] = {};

            if(val === "") delete allWeekPlans[currentWeekOffset][`${day}-${cat}`];

            else allWeekPlans[currentWeekOffset][`${day}-${cat}`] = val;

            saveToStorage();

            updateWeekUI();

        };


        window.addDish = () => {

            const name = document.getElementById('dishName').value.trim();

            const cat = document.getElementById('dishCategory').value;

            if(!name) return;

            dishes.push({id: Date.now(), name, category: cat});

            document.getElementById('dishName').value = '';

            saveToStorage();

            renderFullDishGrid();

            updateWeekUI();

            showToast('Opgeslagen! ✨');

        };


        function renderFullDishGrid() {

            const grid = document.getElementById('fullDishGrid');

            grid.innerHTML = dishes.map(d => `

                <div class="p-4 bg-white rounded-xl border border-orange-100 flex justify-between items-center">

                    <div><span class="text-[10px] font-black text-orange-300 uppercase">${d.category}</span><h3 class="font-bold">${d.name}</h3></div>

                    <button onclick="deleteDish(${d.id})" class="text-red-200 hover:text-red-500 font-bold text-xl">×</button>

                </div>

            `).join('');

        }


        window.deleteDish = (id) => {

            dishes = dishes.filter(d => d.id !== id);

            saveToStorage();

            renderFullDishGrid();

            updateWeekUI();

        }


        function renderShoppingList() {

            const plan = allWeekPlans[currentWeekOffset] || {};

            const meals = [...new Set(Object.values(plan))].filter(Boolean);

            document.getElementById('autoShoppingList').innerHTML = meals.map(m => `

                <li class="flex items-center gap-3 p-3 bg-warm-cream border border-orange-100 rounded-xl shadow-sm">

                    <input type="checkbox" class="w-5 h-5 accent-[#7d8e7a]"> <span class="font-bold">${m}</span>

                </li>

            `).join('');


            document.getElementById('extraShoppingList').innerHTML = extras.map((ex, i) => `

                <li class="flex items-center justify-between p-3 bg-orange-50 rounded-xl">

                    <div class="flex items-center gap-3"><input type="checkbox" class="w-5 h-5 accent-[#7d8e7a]"> <span class="font-bold">${ex}</span></div>

                    <button onclick="removeExtra(${i})" class="text-red-300 font-bold text-xl">×</button>

                </li>

            `).join('');

        }


        window.addExtraItem = () => {

            const input = document.getElementById('extraItem');

            if(!input.value) return;

            extras.push(input.value);

            input.value = '';

            saveToStorage();

            renderShoppingList();

        }


        window.removeExtra = (i) => {

            extras.splice(i, 1);

            saveToStorage();

            renderShoppingList();

        }


        window.clearWeek = () => {

            if(confirm('Week wissen?')) {

                allWeekPlans[currentWeekOffset] = {};

                saveToStorage();

                updateWeekUI();

            }

        }


        function showToast(msg) {

            const t = document.getElementById('toast');

            t.innerText = msg;

            t.classList.remove('translate-y-24');

            setTimeout(() => t.classList.add('translate-y-24'), 2000);

        }


        window.onload = init;

    </script>
